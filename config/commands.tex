% =============================================================================
% Simplifies part preamble creation - no more repetition!
% Usage: \createpart[preamble text]{Part Title}{part:label}
\newcommand{\createpart}[3][]{%
    \ifx&#1&%
        % No preamble provided
        \part{#2}\label{#3}%
    \else%
        % Preamble provided
        \setpartpreamble[c][0.75\linewidth]{%
            \bigskip
            #1%
        }%
        \part{#2}\label{#3}%
    \fi%
}
